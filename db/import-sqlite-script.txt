
PRAGMA foreign_keys = ON;


DROP TABLE IF EXISTS appearance;
DROP TABLE IF EXISTS settings;
DROP TABLE IF EXISTS motivations;
DROP TABLE IF EXISTS abilities;
DROP TABLE IF EXISTS traits;
DROP TABLE IF EXISTS skills;
DROP TABLE IF EXISTS lifepaths;
DROP TABLE IF EXISTS keepsakes;
DROP TABLE IF EXISTS events;
DROP TABLE IF EXISTS opportunities;
DROP TABLE IF EXISTS turning_points;


CREATE TABLE appearance(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL,
   category TEXT NOT NULL,
   description TEXT,
   aging INTEGER NOT NULL,
   scar INTEGER NOT NULL
);

CREATE TABLE settings(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL
);

CREATE TABLE motivations(
   id TEXT PRIMARY KEY,
   low TEXT NOT NULL,
   high TEXT NOT NULL
);

CREATE TABLE abilities(
   id TEXT PRIMARY KEY,
   name TEXT,
   type TEXT
);

CREATE TABLE traits(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL,
   category TEXT NOT NULL,
   source TEXT,
   modifier TEXT,
   description TEXT,
   age INTEGER NOT NULL,
   scar INTEGER NOT NULL
);

CREATE TABLE skills(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL, 
   settings TEXT,
   category TEXT,
   description TEXT
);

CREATE TABLE lifepaths(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL,
   setting TEXT NOT NULL,
   years INTEGER NOT NULL,
   traits TEXT,
   description TEXT,
   baby INTEGER NOT NULL,
   kid INTEGER NOT NULL,
   magic INTEGER NOT NULL,
   guild INTEGER NOT NULL,
   outcast INTEGER NOT NULL
);

CREATE TABLE keepsakes(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL,
   type TEXT NOT NULL,
   description TEXT,
   effect TEXT
);

CREATE TABLE events(
   id TEXT PRIMARY KEY,
   name TEXT NOT NULL,
   prereq TEXT NOT NULL,
   test_skill_id TEXT,
   difficulty INT NOT NULL,
   setup TEXT NOT NULL,
   success_description TEXT,
   success_effect TEXT,
   failure_description TEXT,
   failure_effect TEXT
);

CREATE TABLE opportunities(
   id TEXT PRIMARY KEY,
   prereq TEXT NOT NULL,
   choice_description TEXT NOT NULL,
   motivations TEXT,
   abilities TEXT,
   skills TEXT,
   frequency INTEGER NOT NULL,
   difficulty INTEGER NOT NULL,
   setup TEXT NOT NULL,
   success_description TEXT,
   success_lifepaths TEXT NOT NULL,
   failure_description TEXT,
   failure_lifepaths TEXT NOT NULL
);

CREATE TABLE turning_points(
   id TEXT PRIMARY KEY,
   setting TEXT NOT NULL,
   description TEXT,
   opportunities TEXT
);


.mode csv

.import --skip 1 "idlegamedata/appearance.csv" appearance
.import --skip 1 "idlegamedata/settings.csv" settings
.import --skip 1 "idlegamedata/motivations.csv" motivations
.import --skip 1 "idlegamedata/abilities.csv" abilities
.import --skip 1 "idlegamedata/traits.csv" traits
.import --skip 1 "idlegamedata/skills.csv" skills
.import --skip 1 "idlegamedata/lifepaths.csv" lifepaths
.import --skip 1 "idlegamedata/keepsakes.csv" keepsakes
.import --skip 1 "idlegamedata/events.csv" events
.import --skip 1 "idlegamedata/opportunities.csv" opportunities
.import --skip 1 "idlegamedata/turning_points.csv" turning_points

.quit
